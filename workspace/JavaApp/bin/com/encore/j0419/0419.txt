<속성변수와 참조변수의 차이점>
 속성변수
 - 데이터 값(value)을 담는 변수
 
  값 복사와 주소 복사
  int su1;
  int su2;
  
  su1 = 100;	su1 : 100
  su1 = su2;	su1 : 100, su2 : 100  ----> 값 복사
--------------------------------------------------
  A a1;
  A a2;
  
  a1 = new A();					a1------>203호<-----a2
  a1.su = 100;						   [su=100]
  a2 = a1;
